2linkint n = 0, c=5, frms = 100;

5linkvoid setup() {

6link  size(540, 540);

7link  while (n<500) {

8link    float offSet = PI/500*n;

9link    c = 2;

10link    float a = n*137.5;

11link    float r = c * sqrt(n);

12link    float x = width/2 + cos(a)*r;

13link    float y = height/2 + sin(a)*r;

14link    PVector start = new PVector(x, y);

15link    c = 8;

16link    a = n*137.5;

17link    r = c * sqrt(n);

18link    x = width/2 + cos(a)*r;

19link    y = height/2 + sin(a)*r; 

20link    PVector end = new PVector(x, y);

21link    dots.add(new Dot(start, end, offSet));

22link    n++;

23link  }

24link}

36linkclass Dot {

37link

38link  PVector start, end, v;

39link  float sz = 20;

40link  float x, y, lerpValue, offSet;

41link

42link  Dot(PVector _start, PVector _end, float _offSet) {

43link    start = _start;

44link    end = _end;

45link    offSet = _offSet;

46link  }

47link

48link  void show() {

49link    stroke(238);

50link    fill(34);

51link    ellipse(v.x, v.y, sz, sz);

54link  void update() {

55link    sz = map(sin(theta+offSet), -1, 1, 10, 30);
