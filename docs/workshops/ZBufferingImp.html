La implementación fue realizada en Processing, tomando como base código de 'Nabr' encontrado en los foros de processing.

El código lleva a cabo una implementación del algoritmo de Z-buffering, lo cual permite la representación de figuras tridimensionales a las cuales se les asigna, además de las coordenadas (x,y), una tercera coordenada que representa la profunidad de la figura.

De esta manera, en la implementación se grafican tres esferas con diferentes profundidades (-500, -300 y -100), y se puede apreciar fácilmente cómo se ocultan las partes de las esferas que se encuentran detrás de las otras.

La implementación permite también, graficar una cuarta esfera naranja, la cual tiene una profundidad por defecto de -150, que cambia a -300 al mantener presionada cualquier tecla. Las coordenadas están pensadas de manera que la esfera quede en medio de las esferas roja y verde, y verde y azul respectivamente. La movilidad de la esfera nos permite ver cómo esta interactúa con las otras teniendo en cuenta la profundidad.

Finalmente, si mantenemos presionado el botón izquierdo del mouse, se nos mostrará en pantalla la profundidad correspondiente a la coordenada en la que se encuentra el cursor.

6linkimport java.nio.FloatBuffer;

7linkGL2ES2 gl;

9link//Se declara el buffer con el que se implementará el Z-Buffering

10linkFloatBuffer zbuff;

11linkfloat depth = -150;

13linkvoid setup() {

14link  //Se incializa el espacio tridimensional para poder hacer uso de la coordenada de profundidad.

15link  size(640, 360, P3D);

16link  noStroke();

17link  //Se inicializa el buffer

18link  zbuff = GLBuffers.newDirectFloatBuffer(new float[]{1f});

19link}

21linkvoid draw() {

22link  //Se establecen los valores necesarios para los cálculos del Z-Buffering

23link  float fovy =PI/3.; 

24link  float cameraZ =(height/2.0) / tan(degrees(fovy)*PI / 360.0);  

25link  float zNear =  cameraZ/10.0; 

26link  float zFar=cameraZ*10.0;

27link

28link  //Establecemos el color de fondo, así como la iluminación por defecto

29link  background(255);

30link  lights();

31link

32link  //Dibujamos una esfera con una profundidad de -500 (La de más atrás) -> ROJA

33link  pushMatrix();

34link  translate(width/3.0, height/2.0, -500 );

35link  fill(255,0,0);

36link  sphere(100);

37link  popMatrix();

38link

39link  //Dibujamos una esfera con una profundidad de -300 (La del medio) -> VERDE

40link  pushMatrix();

41link  translate(width/3.0 + 100, height/2.0, -300 );

42link  fill(0,255,0);

43link  sphere(100);

44link  popMatrix();

45link

46link  //Dibujamos una esfera con una profundidad de -100 (La del frente) -> AZUL

47link  pushMatrix();

48link  translate(width/3.0 + 200, height/2.0, -100 );

49link  fill(0,0,255);

50link  sphere(100);

51link  popMatrix();

52link

53link  gl = ((PJOGL) beginPGL()).gl.getGL2ES2();

54link  endPGL();

55link

56link  //Si se mantiene el click, se muestra la profundidad de la coordenada del mismo, sino, se 

57link  //muestra la esfera con  profundidad variable

58link  if(mousePressed){

59link    //Se lee la profundidad de la coordenada en la que se encuentra el mouse y se muestra en pantalla

60link    gl.glReadPixels( mouseX, mouseY, 1, 1, GL2ES2.GL_DEPTH_COMPONENT, GL2ES2.GL_FLOAT, zbuff);

61link    float z = 2.0 * ( zbuff.get(0) - 0.5f);

62link    float worldZ =  2.0f*zFar * zNear / (z*(zFar-zNear)-(zFar+zNear) );

63link    fill(0);

64link    textSize(24);

65link    text(nf(cameraZ+worldZ, 0, 1), mouseX, mouseY);

66link  }else{

67link    //Se cambia la profundidad de la esfera naranja manteniendo presionada cualquier tecla   

68link    if (keyPressed == true) {

69link        depth = -300;

70link    } else {

71link        depth = -150;

72link    }

73link

74link    fill(0);

75link    textSize(18);

76link    text("Orange depth: " + depth, 50, 50);

77link    pushMatrix();

78link    translate(mouseX, mouseY, depth );

79link    fill(255, 165, 0);

80link    sphere(25);

81link    popMatrix();

82link  }  

83link}
